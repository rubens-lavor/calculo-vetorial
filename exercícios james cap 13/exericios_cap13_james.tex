	\item[\textbf{19}]- Determine o vetor tangente unitário $T(t)$ no ponto com valor de parâmetro dado $t$.
	
	$\textbf{r}(t) = \cos(t)\vec{i} + 3t \vec{j} + 2\sin(2t)\vec{k}$
	
	
	\textbf{Resposta:}
	\doublespacing
	

	$\textbf{r}'(t) = -\sin(t)\textbf{i}+3\textbf{j}+4\cos(2t)\textbf{k}$ $\Rightarrow $ $\textbf{r}'(0)=3\textbf{j}+4\textbf{k}$, 
	
	portanto:
	
	$\textbf{T}(0)=\dfrac{\textbf{r}'(0)}{\vert \textbf{r}'(0) \vert} = \dfrac{1}{\sqrt{0^2+3^2+4^2}}(3\textbf{j}+4\textbf{k})=\dfrac{1}{5}(3\textbf{j}+4\textbf{k})= \dfrac{3}{5}\textbf{j}+\dfrac{4}{5}\textbf{k}$
	
	\doublespacing
	
	\item[\textbf{22}] - Se $\textbf{r}(t) = \langle e^{2t},e^{-2t},te^{2t}\rangle$, encontre $\textbf{T}(0),\textbf{r}''(0)$ e $\textbf{r}'(t)\cdot\textbf{r}''(t)$.
	
	\textbf{Resposta:}	
	\doublespacing
	
	$\textbf{r}(t)=\langle e^{2t},e^{-2t},te^{2t} \rangle \Rightarrow \textbf{r}'(t)=\langle 2e^{2t},-2e^{-2t},(2t+1)e^{2t} \rangle \Rightarrow$ 
	
	$\textbf{r}'(0)=\langle 2e^{0},-2e^{0},(0+1)e^0 \rangle = \langle 2,-2,1 \rangle$ e $\vert \textbf{r}'(0) \vert = \sqrt{2^2+(-2)^2+1^2}=3$. 
	
	Então para $\textbf{T}(0)$ temos:
	
	$\textbf{T}(0)=\dfrac{\textbf{r}'(0)}{\vert \textbf{r}'(0) \vert} = \frac{1}{3}\langle 2,-2,1 \rangle = \langle \frac{2}{3}, \frac{-2}{3}, \frac{1}{3} \rangle$
	
	\doublespacing
	
	Para $\textbf{r}''(0)$:
	
	$\textbf{r}''(t)=\langle 4e^{2t},4e^{-2t},(4t + 4)e^{2t} \rangle \Rightarrow \textbf{r}''(0)=\langle 4e^{0},4e^{0},(0 + 4)e^{0} \rangle = \langle 4,4,4 \rangle$

	\doublespacing
	
	Para $\textbf{r}'(t) \cdot \textbf{r}''(t)$:	
	
	$\textbf{r}'(t)\cdot\textbf{r}''(t)=\langle 2e^{2t},-2e^{-2t},(2t+1)e^{2t} \rangle \cdot \langle 4e^{2t},4e^{-2t},(4t + 4)e^{2t} \rangle$
	
	$\quad \quad \quad \quad \quad =$ 
	$(2e^{2t})(4e^{2t})+(-2e^{-2t})(4e^{-2t})+((2t+1)e^{2t})((4t+4)e^{2t})$
	
	$\quad \quad \quad \quad \quad =$ 
	$8e^{4t}-8e^{-4t}+(8t^2+12t+4)e^{4t} = (8t^2+12t+12)e^{4t}-8e^{-4t}$
		
	
	\doublespacing
	\item[\textbf{26}] - Determine as equações paramétricas para a reta tangente à curva dada pelas equações paramétricas, no ponto especificado. 
	$x = \sqrt{t^2 +3}$ ; $y= \ln{t^2 +3}$ ; $z=t$;  $(2,\ln{4},1)$
	
	\textbf{Resposta:}
	\doublespacing
	
	
A equação vetorial para a curva é $\textbf{r}(t)=\langle \sqrt{t^2+3}, \ln{(t^2+3)},t \rangle$, então $\textbf{r}'(t)= \left\langle \dfrac{t}{\sqrt{t^2+3}}, \dfrac{2t}{(t^2+3)},1 \right\rangle$

Em $(2,\ln{4},1)$, $t=1$ e $\textbf{r}'(1)=\left\langle \dfrac{1}{2}, \dfrac{1}{2}, 1 \right\rangle$. 
Portanto, as equações paramétricas da reta tangente são $x=2+ \frac{1}{2}t$, $y=\ln{4} + \frac{1}{2}t$, $z=1+t$


	
	
	\doublespacing
	
	\item[\textbf{27}] - Encontre uma equação para a reta tangente à curva de intersecção dos cilindros $x^{2} + y^{2} = 25$ e $y^{2} + z^{2} = 20$ no ponto $(3,4,2)$.
	
	\textbf{Resposta:}
	\doublespacing
	
	
Primeiro, parametrizamos a curva \textit{C} de interseção. A projeção de \textit{C} no $xy$-plano está contida no círculo $x^2+y^2=25$, $z=0$; então podemos escrever $x=5\cos(t)$, $y=5\sin(t)$. \textit{C} também se encontra no cilindro $y^2+z^2=20$, e $z\geq0$ 
perto do ponto $(3,4,2)$, possamos escrever $z=\sqrt{20-y^2} = \sqrt{20-25\sin^2(t)}$. Uma equação vetorial para \textit{C} é:
	$\textbf{r}(t)=\langle 5\cos(t),5\sin(t),\sqrt{20-25\sin^2(t)} \rangle \Rightarrow \textbf{r}'(t)=\langle-5\sin(t),5\cos(t),\frac{1}{2}(20-25\sin^2(t)^{-1/2})(-50\sin(t)\cos(t))\rangle$
	
	
A reta tangente é paralela a este vetor e passa por $(3,4,2)$, então uma equação vetorial para a reta é

$\textbf{r}(t)=(3-4t)\textbf{i}+(4+3t)\textbf{j}+(2-6t)\textbf{k}$
	\doublespacing
	
	\item[\textbf{28}] - Encontre o ponto na curva de $\textbf{r}(t) = \langle 2\cos(t), 2\sin(t), e^t \rangle$, $0 \leqslant t \leqslant \pi$, em que  a reta tangente é paralela ao plano $\sqrt{3}+y=1$
	
	\textbf{Resposta:}
	\doublespacing
	
	$\textbf{r}(t)=\langle 2\cos(t),2\sin(t),e^t \rangle \Rightarrow \textbf{r}'(t)= \langle -2\sin(t),2\cos(t),e^t \rangle$. 
A reta tangente à curva é paralela ao plano quando o

o vetor tangente da curva é ortogonal ao vetor normal do plano. 

Portanto, exigimos que $\langle -2\sin(t),2\cos(t),e^t \rangle \cdot \langle \sqrt{3},1,0 \rangle = 0 \Rightarrow -2\sqrt{3}\sin(t)+2\cos(t)+0=0$

$\Rightarrow \tg(t)=\frac{1}{\sqrt{3}} \Rightarrow t=\frac{\pi}{6}$ com $[0 \leq t \leq \pi]$

$\textbf{r}(\frac{\pi}{6})=\langle \sqrt{3},1,e^{\pi/6} \rangle$, então o ponto é $(\sqrt{3},1,e^{\pi/6})$.

\end{enumerate}
